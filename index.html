<script>

    /**

     * CẤU HÌNH DÀNH CHO THẦY HOAN

     */

    // 1. Thầy thay đổi danh sách học sinh của mình ở đây:

    const myStudents = ["Nguyệt", "Ngọc Duy", "Linh Đan", "Tommy", "Tra", "Amanda", "Moon"];


    // 2. Thầy dán Link Web App từ Google Apps Script vào đây:

    const GOOGLE_SCRIPT_URL = "DÁN_LINK_CỦA_THẦY_VÀO_ĐÂY";


    // Khởi tạo ngày và danh sách

    document.getElementById('display-date').innerText = new Date().toLocaleDateString('vi-VN');

    document.getElementById('total-count').innerText = myStudents.length;


    const listElement = document.getElementById('list-students');

    myStudents.forEach((name, index) => {

        const row = `

            <tr>

                <td>

                    <div class="student-info">

                        <span class="student-name">${name}</span>

                        <span class="student-sub">ID: ${202600 + index}</span>

                    </div>

                </td>

                <td>

                    <div class="action-group">

                        <button class="btn-status" onclick="set(this, 'Có mặt')">CÓ</button>

                        <button class="btn-status" onclick="set(this, 'Vắng')">VẮNG</button>

                    </div>

                </td>

            </tr>

        `;

        listElement.innerHTML += row;

    });


    // Hàm chọn trạng thái

    function set(el, status) {

        const parent = el.parentElement;

        parent.querySelectorAll('button').forEach(b => b.classList.remove('active-co', 'active-vang'));

        if(status === 'Có mặt') el.classList.add('active-co');

        else el.classList.add('active-vang');

    }


    // Hàm gửi dữ liệu

    async function submitAttendance() {

        const rows = document.querySelectorAll('#list-students tr');

        let data = [];

        

        rows.forEach(r => {

            const name = r.querySelector('.student-name').innerText;

            const activeBtn = r.querySelector('.active-co, .active-vang');

            if(activeBtn) data.push({ name: name, status: activeBtn.innerText });

        });


        if(data.length < myStudents.length) {

            alert("⚠️ Thầy ơi, còn học sinh chưa được điểm danh!");

            return;

        }


        const btn = document.getElementById('main-btn');

        const msg = document.getElementById('msg-box');

        

        btn.disabled = true;

        msg.innerHTML = "⏳ Đang gửi lên Google Sheets...";

        msg.style.color = "#f39c12";


        try {

            await fetch(GOOGLE_SCRIPT_URL, {

                method: 'POST',

                mode: 'no-cors', // Tránh lỗi bảo mật trình duyệt

                body: JSON.stringify(data)

            });

            

            msg.innerHTML = "✅ LƯU THÀNH CÔNG!";

            msg.style.color = "#27ae60";

            alert("Dữ liệu điểm danh đã được gửi vào file Excel của thầy.");

        } catch (e) {

            msg.innerHTML = "❌ LỖI KẾT NỐI!";

            msg.style.color = "#e74c3c";

            console.error(e);

        } finally {

            btn.disabled = false;

        }

    }
